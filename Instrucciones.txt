#HABILITAR EJECUCION DE SCRIPTS TEMPORALMENTE
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass

#Crear un nuevo proyecto con EXPO
npx create-expo-app base --template

#Correr el proyecto
npx expo start -c
npx expo start --web --https

#Ahora corro el proyecto con mi propio Expo personalizado (Dev Client)
npx expo start --dev-client

#COMPILAR APK:
eas build -p android --profile preview

#COMPILAR AAB:
En el eas.json:
    - "buildType": "app-bundle"
eas build -p android --profile production

#Obtener credenciales, SHA1
expo credentials:manager -p android
eas credentials

#Generar build
npx eas build -p android

#Correr el apk:
npx expo run:android

#Tip: Error de APK con EAS
Para generar el APK con EAS, donde hay dependencias nativas, usé:
npx expo install expo-dev-client

#PASOS PARA LOGIN CON GOOGLE
Vídeo: https://www.youtube.com/watch?v=ff9phJxk5ck&ab_channel=Neunapp

1. npx expo install expo-auth-session expo-crypto
2. https://console.cloud.google.com/ > Credentials (android)
    2.1 Añadimos en app.json > Android: "package": "com.douapps.base"
    2.2 npm install -g eas-cli | eas login (Generalmente esto va a estar OK)
    2.3 eas build:configure (crea el eas.json)
    2.4 eas credentials (con esto sacamos el SHA-1)
3. npx expo prebuild --clean
4. npm run android
    4.1 Posible error, ir a: C:\Users\jorge\OneDrive\Escritorio\Proyectos\Ofertapp\base\base\android
    4.2 Creamos un archivo llamado: local.properties
    3.4 Dentro del archivo ponemos: sdk.dir=C:\\Users\\jorge\\AppData\\Local\\Android\\Sdk
